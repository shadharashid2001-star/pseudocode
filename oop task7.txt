

TASK 308 : INSURANCE CLAIM PROCESSING

CLASS CLAIM
    ATTRIBUTES
        ID : NUMBER
        DATE : DATE
        STAGES : TEXT
        DOCUMENTATIONFOLDER : LIST<DOCUMENTATIONFOLDER>

    CONSTRUCTOR(I:NUMBER, D:DATE, ST:TEXT, DOC:LIST<DOCUMENTATIONFOLDER>)
        SET ID TO I
        SET DATE TO D
        SET STAGES TO ST
        SET DOCUMENTATIONFOLDER TO DOC
    END CONSTRUCTOR

    METHODS

        GET_DATE()
            DISPLAY DATE
        END METHOD

        GET_STAGES()
            DISPLAY STAGES
        END METHOD

        CALCULATEPAYOUT()
            DISPLAY "CALCULATE PAYOUT"
        END METHOD

END CLASS


CLASS VEHICLECLAIM INHERITS CLAIM

    ATTRIBUTES
        REPAIRESTIMATE : NUMBER
        FIXED_EXCESS : NUMBER

    CONSTRUCTOR(I:NUMBER,D:DATE,ST:TEXT,DOC:LIST<DOCUMENTATIONFOLDER>,ERP:NUMBER,FIE:NUMBER)
        SUPER(I,D,ST,DOC)
        SET REPAIRESTIMATE TO ERP
        SET FIXED_EXCESS TO FIE
    END CONSTRUCTOR

    METHODS

        GET_REPAIRESTIMATE()
            DISPLAY REPAIRESTIMATE
        END METHOD

        GET_FIXED_EXCESS()
            DISPLAY FIXED_EXCESS
        END METHOD

        OVERRIDE CALCULATEPAYOUT()
            DISPLAY REPAIRESTIMATE - FIXED_EXCESS
        END METHOD

END CLASS


CLASS MEDICALCLAIM INHERITS CLAIM

    ATTRIBUTES
        TREATMENT_COST : NUMBER
        LIMIT : NUMBER

    CONSTRUCTOR(I:NUMBER,D:DATE,ST:TEXT,DOC:LIST<DOCUMENTATIONFOLDER>,TRCOS:NUMBER,LI:NUMBER)
        SUPER(I,D,ST,DOC)
        SET TREATMENT_COST TO TRCOS
        SET LIMIT TO LI
    END CONSTRUCTOR

    METHODS

        GET_TREATMENT_COST()
            DISPLAY TREATMENT_COST
        END METHOD

        GET_LIMIT()
            DISPLAY LIMIT
        END METHOD

        OVERRIDE CALCULATEPAYOUT()
            IF TREATMENT_COST > LIMIT THEN
                DISPLAY LIMIT
            ELSE
                DISPLAY TREATMENT_COST
            END IF
        END METHOD

END CLASS


CLASS PROPERTYCLAIM INHERITS CLAIM

    ATTRIBUTES
        DAMAGE_VALUE : NUMBER
        PERCENTAGE : NUMBER
        INSPECTION_REPORT : TEXT

    CONSTRUCTOR(I:NUMBER,D:DATE,ST:TEXT,DOC:LIST<DOCUMENTATIONFOLDER>,DAM:NUMBER,PER:NUMBER,INSR:TEXT)
        SUPER(I,D,ST,DOC)
        SET DAMAGE_VALUE TO DAM
        SET PERCENTAGE TO PER
        SET INSPECTION_REPORT TO INSR
    END CONSTRUCTOR

    METHODS

        GET_DAMAGE_VALUE()
            DISPLAY DAMAGE_VALUE
        END METHOD

        GET_PERCENTAGE()
            DISPLAY PERCENTAGE
        END METHOD

        GET_INSPECTION_REPORT()
            DISPLAY INSPECTION_REPORT
        END METHOD

        OVERRIDE CALCULATEPAYOUT()
            IF INSPECTION_REPORT = "PROPERLY MAINTAINED" THEN
                DISPLAY DAMAGE_VALUE
            ELSE
                DISPLAY DAMAGE_VALUE - (DAMAGE_VALUE * PERCENTAGE / 100)
            END IF
        END METHOD

END CLASS


CLASS DOCUMENTATIONFOLDER

    ATTRIBUTES
        ID : NUMBER
        NAME : TEXT
        PATH : TEXT

    CONSTRUCTOR(I:NUMBER,N:TEXT,PTH:TEXT)
        SET ID TO I
        SET NAME TO N
        SET PATH TO PTH
    END CONSTRUCTOR

    METHODS

        GET_NAME()
            DISPLAY NAME
        END METHOD

        GET_PATH()
            DISPLAY PATH
        END METHOD

END CLASS


CLASS POLICY

    ATTRIBUTES
        CLAIMS : LIST<CLAIM>

    CONSTRUCTOR()
        CREATE EMPTY LIST CLAIMS
    END CONSTRUCTOR

    METHODS

        ADD_CLAIM(NEWCLAIM : CLAIM)
            ADD NEWCLAIM TO CLAIMS
        END METHOD

        DISPLAY_CLAIMS()
            FOR EACH C IN CLAIMS
                DISPLAY C
            END FOR
        END METHOD

END CLASS

CLASS JOURNALISSUE INHERITS REFERENCETEXTBOOK

ATTRIBUTES

RENEWABLE : BOOLEAN

CONSTRUCTOR(T:TEXT, BA:TEXT, LO:NUMBER, CR:CONDITIONREPORT, LA:NUMBER, RE:BOOLEAN)
    SUPER(T, BA, LO, CR, LA)
    SET RENEWABLE TO RE
END CONSTRUCTOR

METHODS

METHOD CalculateLateFee(DaysLate:NUMBER)
    DISPLAY DaysLate * LATEFEEPERDAY
END METHOD

METHOD Renew(DaysLate:NUMBER)
    IF DaysLate GREATER THAN 0 THEN
        SET RENEWABLE TO FALSE
        DISPLAY "CANNOT RENEW"
    ELSE
        DISPLAY "RENEWED"
    END IF
END METHOD

END CLASS
