CLASS CLAIM
    ATTRIBUTES
        ID : NUMBER
        DATE : DATE
        STAGES : TEXT
        DOCUMENTATIONFOLDER : LIST<DOCUMENTATIONFOLDER>

    CONSTRUCTOR(I:NUMBER, D:DATE, ST:TEXT, DOC:LIST<DOCUMENTATIONFOLDER>)
        SET ID TO I
        SET DATE TO D
        SET STAGES TO ST
        SET DOCUMENTATIONFOLDER TO DOC
    END CONSTRUCTOR

    METHODS

        GET_DATE()
            DISPLAY DATE
        END METHOD

        GET_STAGES()
            DISPLAY STAGES
        END METHOD

        CALCULATEPAYOUT()
            DISPLAY "CALCULATE PAYOUT"
        END METHOD

END CLASS


CLASS VEHICLECLAIM INHERITS CLAIM

    ATTRIBUTES
        REPAIRESTIMATE : NUMBER
        FIXED_EXCESS : NUMBER

    CONSTRUCTOR(I:NUMBER,D:DATE,ST:TEXT,DOC:LIST<DOCUMENTATIONFOLDER>,ERP:NUMBER,FIE:NUMBER)
        SUPER(I,D,ST,DOC)
        SET REPAIRESTIMATE TO ERP
        SET FIXED_EXCESS TO FIE
    END CONSTRUCTOR

    METHODS

        GET_REPAIRESTIMATE()
            DISPLAY REPAIRESTIMATE
        END METHOD

        GET_FIXED_EXCESS()
            DISPLAY FIXED_EXCESS
        END METHOD

        OVERRIDE CALCULATEPAYOUT()
            DISPLAY REPAIRESTIMATE - FIXED_EXCESS
        END METHOD

END CLASS


CLASS MEDICALCLAIM INHERITS CLAIM

    ATTRIBUTES
        TREATMENT_COST : NUMBER
        LIMIT : NUMBER

    CONSTRUCTOR(I:NUMBER,D:DATE,ST:TEXT,DOC:LIST<DOCUMENTATIONFOLDER>,TRCOS:NUMBER,LI:NUMBER)
        SUPER(I,D,ST,DOC)
        SET TREATMENT_COST TO TRCOS
        SET LIMIT TO LI
    END CONSTRUCTOR

    METHODS

        GET_TREATMENT_COST()
            DISPLAY TREATMENT_COST
        END METHOD

        GET_LIMIT()
            DISPLAY LIMIT
        END METHOD

        OVERRIDE CALCULATEPAYOUT()
            IF TREATMENT_COST > LIMIT THEN
                DISPLAY LIMIT
            ELSE
                DISPLAY TREATMENT_COST
            END IF
        END METHOD

END CLASS


CLASS PROPERTYCLAIM INHERITS CLAIM

    ATTRIBUTES
        DAMAGE_VALUE : NUMBER
        PERCENTAGE : NUMBER
        INSPECTION_REPORT : TEXT

    CONSTRUCTOR(I:NUMBER,D:DATE,ST:TEXT,DOC:LIST<DOCUMENTATIONFOLDER>,DAM:NUMBER,PER:NUMBER,INSR:TEXT)
        SUPER(I,D,ST,DOC)
        SET DAMAGE_VALUE TO DAM
        SET PERCENTAGE TO PER
        SET INSPECTION_REPORT TO INSR
    END CONSTRUCTOR

    METHODS

        GET_DAMAGE_VALUE()
            DISPLAY DAMAGE_VALUE
        END METHOD

        GET_PERCENTAGE()
            DISPLAY PERCENTAGE
        END METHOD

        GET_INSPECTION_REPORT()
            DISPLAY INSPECTION_REPORT
        END METHOD

        OVERRIDE CALCULATEPAYOUT()
            IF INSPECTION_REPORT = "PROPERLY MAINTAINED" THEN
                DISPLAY DAMAGE_VALUE
            ELSE
                DISPLAY DAMAGE_VALUE - (DAMAGE_VALUE * PERCENTAGE / 100)
            END IF
        END METHOD

END CLASS


CLASS DOCUMENTATIONFOLDER

    ATTRIBUTES
        ID : NUMBER
        NAME : TEXT
        PATH : TEXT

    CONSTRUCTOR(I:NUMBER,N:TEXT,PTH:TEXT)
        SET ID TO I
        SET NAME TO N
        SET PATH TO PTH
    END CONSTRUCTOR

    METHODS

        GET_NAME()
            DISPLAY NAME
        END METHOD

        GET_PATH()
            DISPLAY PATH
        END METHOD

END CLASS


CLASS POLICY

    ATTRIBUTES
        CLAIMS : LIST<CLAIM>

    CONSTRUCTOR()
        CREATE EMPTY LIST CLAIMS
    END CONSTRUCTOR

    METHODS

        ADD_CLAIM(NEWCLAIM : CLAIM)
            ADD NEWCLAIM TO CLAIMS
        END METHOD

        DISPLAY_CLAIMS()
            FOR EACH C IN CLAIMS
                DISPLAY C
            END FOR
        END METHOD

END CLASS


START

BEGIN

    CREATE EMPTY LIST DOCS

    CREATE OBJECT DOC1 TO DOCUMENTATIONFOLDER(1,"Police Report","C:\Docs\Police.pdf")
    ADD DOC1 TO DOCS

    CREATE OBJECT CLAIM1 TO VEHICLECLAIM(101,"01/05/2026","Submitted",DOCS,5000,500)

    CREATE OBJECT CLAIM2 TO MEDICALCLAIM(102,"02/05/2026","Approved",DOCS,3500,4000)

    CREATE OBJECT CLAIM3 TO PROPERTYCLAIM(103,"03/05/2026","Processing",DOCS,12000,20,"PROPERLY MAINTAINED")

    CREATE OBJECT POLICY1 TO POLICY()

    POLICY1.ADD_CLAIM(CLAIM1)
    POLICY1.ADD_CLAIM(CLAIM2)
    POLICY1.ADD_CLAIM(CLAIM3)

    CLAIM1.CALCULATEPAYOUT()
    CLAIM2.CALCULATEPAYOUT()
    CLAIM3.CALCULATEPAYOUT()

END

CLASS FIXEDPRICEINVOICE INHERITS INVOICE

    SUPER(IN,CN,DA,PT)
    SET PROJECTFEE TO PF
END CONSTRUCTOR

METHODS

 CalculateTotal()
    DISPLAY PROJECTFEE
END METHOD

END CLASS


CLASS HOURLYINVOICE INHERITS INVOICE

ATTRIBUTES

HOURLYRATE : NUMBER
HOURSWORKED : NUMBER
APPROVED : BOOLEAN

CONSTRUCTOR(IN:TEXT,CN:TEXT,DA:TEXT,PT:PAYMENTTERMS,HR:NUMBER,HW:NUMBER,AP:BOOLEAN)
    SUPER(IN,CN,DA,PT)
    SET HOURLYRATE TO HR
    SET HOURSWORKED TO HW
    SET APPROVED TO AP
END CONSTRUCTOR

METHODS

 CalculateTotal()
    IF APPROVED EQUALS TRUE THEN
        DISPLAY HOURLYRATE * HOURSWORKED
    ELSE
        DISPLAY "HOURS NOT APPROVED"
    END IF
END METHOD

END CLASS


CLASS RETAINERINVOICE INHERITS FIXEDPRICEINVOICE

ATTRIBUTES

INCLUDEDHOURS : NUMBER
EXTRAHOURS : NUMBER
EXTRAHOURLYRATE : NUMBER

CONSTRUCTOR(IN:TEXT,CN:TEXT,DA:TEXT,PT:PAYMENTTERMS,PF:NUMBER,IH:NUMBER,EH:NUMBER,ER:NUMBER)
    SUPER(IN,CN,DA,PT,PF)
    SET INCLUDEDHOURS TO IH
    SET EXTRAHOURS TO EH
    SET EXTRAHOURLYRATE TO ER
END CONSTRUCTOR

METHODS

 CalculateTotal()
    DISPLAY PROJECTFEE + (EXTRAHOURS * EXTRAHOURLYRATE)
END METHOD

END CLASS
