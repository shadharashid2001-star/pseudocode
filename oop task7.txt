        SUPER(I,D,ST,DOC)
        SET DAMAGE_VALUE TO DAM
        SET PERCENTAGE TO PER
        SET INSPECTION_REPORT TO INSR
    END CONSTRUCTOR

        GET_DAMAGE_VALUE()
            DISPLAY DAMAGE_VALUE
        END METHOD

        GET_PERCENTAGE()
            DISPLAY PERCENTAGE
        END METHOD

        GET_INSPECTION_REPORT()
            DISPLAY INSPECTION_REPORT
        END METHOD

        OVERRIDE CALCULATEPAYOUT()
            IF INSPECTION_REPORT = "PROPERLY MAINTAINED" THEN
                DISPLAY DAMAGE_VALUE
            ELSE
                DISPLAY DAMAGE_VALUE - (DAMAGE_VALUE * PERCENTAGE / 100)
            END IF
        END METHOD

            END IF

CLASS DATABASESERVER INHERITS SERVER

    ATTRIBUTES
        QUERYTHROUGHPUT : NUMBER
        MAXQPS : NUMBER

    CONSTRUCTOR(SN:TEXT,UP:NUMBER,HS:HARDWARESPECIFICATION,
                QT:NUMBER,MQ:NUMBER)

        SUPER(SN,UP,HS)
        SET QUERYTHROUGHPUT TO QT
        SET MAXQPS TO MQ

    END CONSTRUCTOR

    METHODS

        OVERRIDE CalculateLoad()
            DISPLAY (QUERYTHROUGHPUT * 100) / MAXQPS
        END METHOD

        AcceptConnection()

            IF QUERYTHROUGHPUT >= MAXQPS THEN
                DISPLAY "CONNECTION REFUSED"
            ELSE
                DISPLAY "CONNECTION ACCEPTED"
            END IF

        END METHOD

END CLASS


CLASS CACHESERVER INHERITS DATABASESERVER

    ATTRIBUTES
        CACHEHITRATIO : NUMBER

    CONSTRUCTOR(SN:TEXT,UP:NUMBER,HS:HARDWARESPECIFICATION,
                QT:NUMBER,MQ:NUMBER,CHR:NUMBER)

        SUPER(SN,UP,HS,QT,MQ)
        SET CACHEHITRATIO TO CHR

    END CONSTRUCTOR

    METHODS

        OVERRIDE CalculateLoad()

            DISPLAY ((QUERYTHROUGHPUT * 100) / MAXQPS)
                    * ((100 - CACHEHITRATIO) / 100)

        END METHOD

END CLASS
