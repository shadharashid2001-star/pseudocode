CLASS ELECTRICITYBILL INHERITS BILL


ATTRIBUTES

PEAK_UNITS : NUMBER
OFFPEAK_UNITS : NUMBER

OFFPEAK_RATE : NUMBER


CONSTRUCTOR(I:NUMBER,BILP:NUMBER,DUDATE:DATE,
M:METERREADING,
PU:NUMBER,OPU:NUMBER,PR:NUMBER,OPR:NUMBER)


    SUPER(I,BILP,DUDATE,M)

    SET PEAK_UNITS TO PU
    SET OFFPEAK_UNITS TO OPU
    SET PEAK_RATE TO PR
    SET OFFPEAK_RATE TO OPR

    RETURN OFFPEAK_UNITS

CLASS GASBILL INHERITS BILL


ATTRIBUTES

UNITS_CONSUMED : NUMBER
RATE : NUMBER
SEASON : TEXT


CONSTRUCTOR(I:NUMBER,BILP:NUMBER,DUDATE:DATE,
M:METERREADING,
UNI:NUMBER,RA:NUMBER,SE:TEXT)


    SUPER(I,BILP,DUDATE,M)

    SET UNITS_CONSUMED TO UNI
    SET RATE TO RA
    SET SEASON TO SE


END CONSTRUCTOR



METHODS


OVERRIDE CALCULATEAMOUNTOWED()


    SET AMOUNT TO UNITS_CONSUMED * RATE


    IF SEASON EQUAL "WINTER" THEN

        SET AMOUNT TO AMOUNT + (AMOUNT * 10 / 100)

    END IF


    RETURN AMOUNT


END METHOD


GET_UNITS_CONSUMED()

    RETURN UNITS_CONSUMED

END METHOD


GET_RATE()

    RETURN RATE

END METHOD


GET_SEASON()

    RETURN SEASON

END METHOD


END CLASS
