CLASS ELECTRICITYBILL INHERITS BILL

  ATTRIBUTES
    UNITS_POWER : NUMBER
    RATE : NUMBER
    TIME : NUMBER

  CONSTRUCTOR(I:NUMBER,BILP:NUMBER,DUDATE:DATE,
              M:METERREADING,
              UNP:NUMBER,RA:NUMBER,TI:NUMBER)

    SUPER(I,BILP,DUDATE,M)

    SET UNITS_POWER TO UNP
    SET RATE TO RA
    SET TIME TO TI

  END CONSTRUCTOR

  METHODS

    OVERRIDE CALCULATEAMOUNTOWED()
      RETURN UNITS_POWER * (RATE * TIME)
    END METHOD

    GET_UNITS_POWER()
      RETURN UNITS_POWER
    END METHOD

    GET_RATE()
      RETURN RATE
    END METHOD

    GET_TIME()
      RETURN TIME
    END METHOD

END CLASS



CLASS WATERBILL INHERITS BILL

  ATTRIBUTES
    LITERS_CONSUMED : NUMBER
    FLAT_RATE : NUMBER
    FIXEDSTANDING : NUMBER

  CONSTRUCTOR(I:NUMBER,BILP:NUMBER,DUDATE:DATE,
              M:METERREADING,
              LI:NUMBER,FRA:NUMBER,FIX:NUMBER)

    SUPER(I,BILP,DUDATE,M)

    SET LITERS_CONSUMED TO LI
    SET FLAT_RATE TO FRA
    SET FIXEDSTANDING TO FIX

  END CONSTRUCTOR

  METHODS

    OVERRIDE CALCULATEAMOUNTOWED()
      RETURN (LITERS_CONSUMED * FLAT_RATE) + FIXEDSTANDING
    END METHOD

    GET_LITERS_CONSUMED()
      RETURN LITERS_CONSUMED
    END METHOD

    GET_FLAT_RATE()
      RETURN FLAT_RATE
    END METHOD

    GET_FIXEDSTANDING()
      RETURN FIXEDSTANDING
    END METHOD

END CLASS



CLASS GASBILL INHERITS BILL

  ATTRIBUTES
    UNITS_CONSUMED : NUMBER
    RATE : NUMBER
    SEASON : TEXT

  CONSTRUCTOR(I:NUMBER,BILP:NUMBER,DUDATE:DATE,
              M:METERREADING,
              UNI:NUMBER,RA:NUMBER,SE:TEXT)

    SUPER(I,BILP,DUDATE,M)

    SET UNITS_CONSUMED TO UNI
    SET RATE TO RA
    SET SEASON TO SE

  END CONSTRUCTOR

  METHODS

    OVERRIDE CALCULATEAMOUNTOWED()

      IF SEASON EQUAL "WINTER" THEN
        RETURN (UNITS_CONSUMED * RATE) + 5
      ELSE
        RETURN (UNITS_CONSUMED * RATE)
      END IF

    END METHOD

    GET_UNITS_CONSUMED()
      RETURN UNITS_CONSUMED
    END METHOD

    GET_RATE()
      RETURN RATE
    END METHOD

    GET_SEASON()
      RETURN SEASON
    END METHOD

END CLASS

Task 307: Ride-Sharing Vehicle Tiers
CLASS Ride

ATTRIBUTES
     ID : NUMBER
     PickupLocation : TEXT
     DropoffLocation : TEXT
     Distance : NUMBER
     Route : ROUTE

CONSTRUCTOR(PU:TEXT,DR:TEXT,DIS:NUMBER,R:ROUTE)
    SET PickupLocation TO PU
    SET DropoffLocation TO DR
    SET Distance TO DIS
    SET Route TO R
END CONSTRUCTOR

METHOD CalculateFare()
END METHOD

METHOD GetDetails()
    DISPLAY PickupLocation
    DISPLAY DropoffLocation
    DISPLAY Distance
END METHOD

END CLASS


CLASS StandardRide INHERITS Ride

ATTRIBUTES
     BaseRate : NUMBER

CONSTRUCTOR(PU:TEXT,DR:TEXT,DIS:NUMBER,R:ROUTE,BR:NUMBER)
    SUPER(PU,DR,DIS,R)
    SET BaseRate TO BR
END CONSTRUCTOR

METHOD CalculateFare()
    DISPLAY Distance * BaseRate
END METHOD

END CLASS


CLASS PremiumRide INHERITS Ride

ATTRIBUTES
     PremiumRate : NUMBER
    ComfortSurcharge : NUMBER

CONSTRUCTOR(PU:TEXT,DR:TEXT,DIS:NUMBER,R:ROUTE,PR:NUMBER,CS:NUMBER)
    SUPER(PU,DR,DIS,R)
    SET PremiumRate TO PR
    SET ComfortSurcharge TO CS
END CONSTRUCTOR

METHOD CalculateFare()
    DISPLAY (Distance * PremiumRate) + ComfortSurcharge
END METHOD

END CLASS


CLASS SharedRide INHERITS StandardRide

ATTRIBUTES
     PassengerCount : NUMBER

CONSTRUCTOR(PU:TEXT,DR:TEXT,DIS:NUMBER,R:ROUTE,BR:NUMBER,PC:NUMBER)
    SUPER(PU,DR,DIS,R,BR)
    SET PassengerCount TO PC
END CONSTRUCTOR

METHOD CalculateFare()
    DISPLAY (Distance * BaseRate) / PassengerCount
END METHOD

END CLASS


CLASS Route

     ID : NUMBER
     Waypoints : LIST<WAYPOINT>

CONSTRUCTOR()
    SET Waypoints TO EMPTY LIST
END CONSTRUCTOR

METHOD AddWaypoint(W:WAYPOINT)
    ADD W TO Waypoints
END METHOD

METHOD DisplayRoute()

END METHOD

END CLASS
