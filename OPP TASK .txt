CLASS PRODUCT
    ATTRIBUTES
        PRIVATE ID: NUMBER
        PRIVATE NAME: TEXT
        PRIVATE CATEGORY: CATEGORY
        PRIVATE BINS: LIST<STORAGEBIN>
        PRIVATE VENDOR: VENDOR

    CONSTRUCTOR(N: TEXT, C: CATEGORY, V: VENDOR)
        SET NAME TO N
        SET CATEGORY TO C
        SET VENDOR TO V
        SET BINS TO EMPTY LIST
    END CONSTRUCTOR

    METHODS
        ADD_BIN(B: STORAGEBIN)
            ADD B TO BINS
        END METHOD

        CALCULATE_TOTAL_STOCK()
            SET TOTAL TO 0
			REPEAT FROM I = 1 TO  SIZE OF BINS
				SET TOTAL TO TOTAL + BINS[I].GET_QUANTITY()
				SET I TO I + 1
			END REPEAT
			DISPLAY TOTAL
        END METHOD

        PROCESS_MOVEMENT(M: STOCKMOVEMENT)
            IF M.TYPE EQUALS TO "OUT" AND M.QUANTITY GREATER THAN PRODUCT.CALCULATE_TOTAL_STOCK() THEN
                DISPLAY "ERROR: INSUFFICIENT STOCK"
            ELSE
                IF M.TYPE EQUALS TO "IN" THEN
                    M.BIN.ADD_QUANTITY(M.QUANTITY)
                ELSE
                    M.BIN.REMOVE_QUANTITY(M.QUANTITY)
                END IF
              ELSE
                DISPLAY "ERROR: BIN DOES NOT BELONG TO PRODUCT"
            END IF
        END METHOD
END CLASS
