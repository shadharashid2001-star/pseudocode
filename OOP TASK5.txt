    SET TOTAL TO 0

    REPEAT FROM I = 1 TO SIZE OF ITEMS

    DISPLAY OUTSTANDINGBALANCE

    IF OUTSTANDINGBALANCE + AMOUNT LESS THAN OR EQUAL TO CREDITLIMIT THEN

        SET OUTSTANDINGBALANCE TO OUTSTANDINGBALANCE + AMOUNT

        DISPLAY "PURCHASE APPROVED"

    ELSE

        DISPLAY "CREDIT LIMIT EXCEEDED"

    END IF

        SET OUTSTANDINGBALANCE TO OUTSTANDINGBALANCE - AMOUNT

        DISPLAY "PAYMENT SUCCESSFUL"

    ELSE

        DISPLAY "INVALID PAYMENT"

    END IF
