Task 296: Online Payment Methods 
 CLASS PAYMENTMETHOD 
  ATTRIBUTES
   ID : NUMBER
   AMOUNT:NUMBER
   STATUS : TEXT 
 CONSTRUCTOR(A:NUMBER,S:TEXT)
    SET AMOUNT TO A
    SET STATUS TO S
END CONSTRUCTOR

METHODS

GET_AMOUNT()
    DISPLAY AMOUNT
END METHOD

GET_STATUS()
    DISPLAY STATUS
END METHOD

PROCESS_PAYMENT()
    DISPLAY "PROCESSING PAYMENT"
END METHOD

END CLASS

CLASS CREDITCARDPAYMENT INHERITS PAYMENTMETHOD
  ATTRIBUTES
    CARDNUM:NUMBER
	EXPIREDAY: TEXT
 CONSTRUCTOR(A:NUMBER,S:TEXT,C:NUMBER, EX:TEXT)
  PAYMENTMETHOD( A,S)
  SET CARDNUM TO C
  SET EXPIREDAY TO EX 
 END CONSTRUCTOR
  
 METHODS
 
 GET_CARDNUM()
   DISPLAY CARDNUM
 END METHOD
 
 GET_EXPIREDAY()
   DISPLAY EXPIREDAY 
 END METHOD
 
PROCESS_PAYMENT()

    IF CARDNUM GREATER THAN 0 AND EXPIREDAY NOT EQUALS "" THEN
        DISPLAY "CARD VALIDATED"
        SET STATUS TO "COMPLETED"
    ELSE
        DISPLAY "INVALID CARD"
    END IF

END METHOD

END CLASS

CLASS BANKTRANSFERPAYMENT INHERITS PAYMENTMETHOD
 ATTRIBUTES 
   ACCOUNTNUM: NUMBER
 CONSTRUCTOR(A:NUMBER,S:TEXT,AN:NUMBER)
   PAYMENTMETHOD(A,S)
   SET ACCOUNTNUM TO AN
 END CONSTRUCTOR
 
METHODS
 
PROCESS_PAYMENT()
  IF ACCOUNTNUM GREATER THAN 0 THEN 
    SET STATUS TO "COMPLETED"
    DISPLAY 'ACCOUNTNUM VALIDATED'
  ELSE
    DISPLAY "INVALID ACCOUNT"
  END IF
 END METHOD
 END CLASS
 
 CLASS WALLETPAYMENT INHERITS PAYMENTMETHOD
  ATTRIBUTES
   BALANCE: NUMBER
   
 CONSTRUCTOR(A:NUMBER,S:TEXT,B:NUMBER)
   PAYMENTMETHOD (A,S)
   SET BALANCE TO B
  END CONSTRUCTOR
  
METHODS
GET_BALANCE()
    DISPLAY BALANCE
END METHOD

PROCESS_PAYMENT()

    IF BALANCE GREATER THAN OR EQUAL TO AMOUNT THEN
        SET BALANCE TO BALANCE - AMOUNT
        SET STATUS TO "COMPLETED"
    ELSE
        DISPLAY "INSUFFICIENT BALANCE"
    END IF

END METHOD

END CLASS

CLASS ORDER

ATTRIBUTES
ID : NUMBER
ITEMS : LIST<ITEM>
PAYMENT : PAYMENTMETHOD

CONSTRUCTOR()
    SET ITEMS TO EMPTY LIST
END CONSTRUCTOR

METHODS

ADD_ITEM(I:ITEM)
    ADD I TO ITEMS
END METHOD

SET_PAYMENT(P:PAYMENTMETHOD)
    SET PAYMENT TO P
END METHOD

CHECKOUT()

    SET TOTAL TO 0

    REPEAT FROM I = 1 TO SIZE OF ITEMS
        SET TOTAL TO TOTAL + ITEMS[I].PRICE
    END REPEAT

    SET PAYMENT.AMOUNT TO TOTAL

    PAYMENT.PROCESS_PAYMENT()

END METHOD

END CLASS

  CREATE OBJECT OR1 OF ORDER()

  OR1.ADD_ITEM(ITEM1)

  OR1.SET_PAYMENT(CARDPAYMENT1)
  OR1.CHECKOUT()
