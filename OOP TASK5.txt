Task 293: Shape Drawing App 
CLASS SHAPE

 ATTRIBUTES
     ID : NUMBER
     COLOR : TEXT
     POSITIONX : NUMBER
     POSITIONY : NUMBER

 CONSTRUCTOR(C:TEXT, X:NUMBER, Y:NUMBER)
    SET COLOR TO C
    SET POSITIONX TO X
    SET POSITIONY TO Y
 END CONSTRUCTOR
METHODS

GET_COLOR()
    DISPLAY COLOR
END METHOD

GET_POSITION()
    DISPLAY POSITIONX
    DISPLAY POSITIONY
END METHOD

GET_AREA()
    DISPLAY "FIND AREA  "
END METHOD
GET_PERIMETER()
    DISPLAY "FIND PERIMETER "
END METHOD

END CLASS 

CLASS CIRCLE INHERITS SHAPE 
 ATTRIBUTES 
  RADIUS : NUMBER
 CONSTRUCTOR(C:TEXT, X:NUMBER, Y:NUMBER,R:NUMBER)
   SHAPE (C,X,Y)
  SET RADIUS TO R 
 END CONSTRUCTOR 
  
 METHODS
 
GET_AREA()
 DISPLAY 3.14 * (RADIUS * RADIUS)
 END METHOD

GET_PERIMETER()
   DISPLAY 2 * ( 3.14 * RADIUS) 
END METHOD

END CLASS

CLASS RECTANGLE INHERITS SHAPE

 ATTRIBUTES
    PRIVATE WIDTH : NUMBER
    PRIVATE HEIGHT : NUMBER

 CONSTRUCTOR(C:TEXT, X:NUMBER, Y:NUMBER, W:NUMBER, H:NUMBER)
    SHAPE(C, X, Y)
    SET WIDTH TO W
    SET HEIGHT TO H
 END CONSTRUCTOR

METHODS

GET_AREA()
    DISPLAY (WIDTH * HEIGHT)
END METHOD

GET_PERIMETER()
    DISPLAY 2 * (WIDTH + HEIGHT)
END METHOD
END CLASS

  SET SIDE3 TO S3

CLASS CANVAS

ATTRIBUTES
     ID : NUMBER
     NAME : TEXT
     SHAPES : LIST<SHAPE>

CONSTRUCTOR(N:TEXT)
    SET NAME TO N
    SET SHAPES TO LIST SHAPE
END CONSTRUCTOR

METHODS

ADD_SHAPE(S:SHAPE)
    ADD S TO SHAPES
END METHOD

DISPLAY_SHAPES()
    REPEAT FROM I = 1 TO SIZE OF SHAPES
        DISPLAY SHAPES[I]
      END REPEAT
END METHOD

END CLASS

     EMPLOYEE(N, H)
    SET MONTHLYSALARY TO M
END CONSTRUCTOR

METHODS

CALCULATEPAY()
    DISPLAY MONTHLYSALARY

END CLASS

CLASS HOURLYEMPLOYEE INHERITS EMPLOYEE

ATTRIBUTES
     HOURLYRATE : NUMBER
     HOURSWORKED : NUMBER

CONSTRUCTOR(N:TEXT, H:TEXT, HR:NUMBER, HW:NUMBER)
    EMPLOYEE(N, H)
    SET HOURLYRATE TO HR
    SET HOURSWORKED TO HW
END CONSTRUCTOR

METHODS

CALCULATEPAY()
    DISPLAY HOURLYRATE * HOURSWORKED
END METHOD

END CLASS

CLASS COMMISSIONEMPLOYEE INHERITS EMPLOYEE

ATTRIBUTES
     BASEPAY : NUMBER
     SALESAMOUNT : NUMBER
     COMMISSIONRATE : NUMBER

CONSTRUCTOR(N:TEXT, H:TEXT, BP:NUMBER, SA:NUMBER, CR:NUMBER)
     EMPLOYEE(N, H)
    SET BASEPAY TO BP
    SET SALESAMOUNT TO SA
    SET COMMISSIONRATE TO CR
END CONSTRUCTOR

METHODS

CALCULATEPAY()
    DISPLAY BASEPAY + (SALESAMOUNT * COMMISSIONRATE / 100)
END METHOD

END CLASS

CLASS PAYROLL

ATTRIBUTES
   ID : NUMBER
    EMPLOYEES : LIST<EMPLOYEE>

CONSTRUCTOR()
    SET EMPLOYEES TO EMPTY LIST
END CONSTRUCTOR

METHODS

ADD_EMPLOYEE(E:EMPLOYEE)
    ADD E TO EMPLOYEES
END METHOD

DISPLAY_EMPLOYEES()
    REPEAT FROM I = 1 TO SIZE OF EMPLOYEES
        DISPLAY EMPLOYEES[I]
    END REPEAT
END METHOD

END CLASS
