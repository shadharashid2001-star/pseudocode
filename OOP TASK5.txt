CLASS CREDITCARDPAYMENT INHERITS PAYMENTMETHOD
  ATTRIBUTES
    CARDNUM:NUMBER
	EXPIREDAY: TEXT
 CONSTRUCTOR(A:NUMBER,S:TEXT,C:NUMBER, EX:TEXT)
  PAYMENTMETHOD( A,S)
  SET CARDNUM TO C
  SET EXPIREDAY TO EX 
 END CONSTRUCTOR
  
 METHODS
 
 GET_CARDNUM()
   DISPLAY CARDNUM
 END METHOD
 
 GET_EXPIREDAY()
   DISPLAY EXPIREDAY 
 END METHOD
 
PROCESS_PAYMENT()

    IF CARDNUM GREATER THAN 0 AND EXPIREDAY NOT EQUALS "" THEN
        DISPLAY "CARD VALIDATED"
        SET STATUS TO "COMPLETED"
    ELSE
        DISPLAY "INVALID CARD"
    END IF

END METHOD

CLASS ORDER

ATTRIBUTES
ID : NUMBER
ITEMS : LIST<ITEM>
PAYMENT : PAYMENTMETHOD

CONSTRUCTOR()
    SET ITEMS TO EMPTY LIST
END CONSTRUCTOR

METHODS

ADD_ITEM(I:ITEM)
    ADD I TO ITEMS
END METHOD

SET_PAYMENT(P:PAYMENTMETHOD)
    SET PAYMENT TO P
END METHOD

CHECKOUT()

    SET TOTAL TO 0

    REPEAT FROM I = 1 TO SIZE OF ITEMS
        SET TOTAL TO TOTAL + ITEMS[I].PRICE
    END REPEAT

    SET PAYMENT.AMOUNT TO TOTAL

    PAYMENT.PROCESS_PAYMENT()

END METHOD

END CLASS

  SET CAPACTIY TO CA 
  SET USEDSPACE TO US 
 END CONSTRUCTOR
 
METHODS

ADD_RECORDING(SIZE:NUMBER)

    IF USEDSPACE + SIZE LESS THAN OR EQUAL TO CAPACITY THEN
        SET USEDSPACE TO USEDSPACE + SIZE
        DISPLAY "RECORDING SAVED"
    ELSE
        DISPLAY "STORAGE FULL"
    END IF

END METHOD
